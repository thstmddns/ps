N, M = map(int, input().split())
arr = list(input() for _ in range(N))
min_cnt = 64
for i in range(N - 8 + 1):
    for j in range(M - 8 + 1):
        cnt1 = 0
        cnt2 = 0
        for r in range(8):
            for c in range(8):
                if (i + j + r + c) % 2:
                    if arr[i + r][j + c] == 'B':
                        cnt1 += 1
                    else:
                        cnt2 += 1
                else:
                    if arr[i + r][j + c] == 'W':
                        cnt1 += 1
                    else:
                        cnt2 += 1
        min_cnt = min(min_cnt, min(cnt1, cnt2))
print((min_cnt))
