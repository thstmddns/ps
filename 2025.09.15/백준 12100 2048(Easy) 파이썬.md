import sys, copy

N = int(sys.stdin.readline())
arr = [list(map(int, sys.stdin.readline().split())) for _ in range(N)]
res = 0

def left(copy_arr):
    for i in range(N):
        cursor = 0
        for j in range(1, N):
            if copy_arr[i][j] != 0:
                temp = copy_arr[i][j]
                copy_arr[i][j] = 0

                if copy_arr[i][cursor] == 0:
                    copy_arr[i][cursor] = temp
                
                elif copy_arr[i][cursor] == temp:
                    copy_arr[i][cursor] += temp
                    cursor += 1
                
                else:
                    cursor += 1
                    copy_arr[i][cursor] = temp

    return copy_arr

def right(copy_arr):
    for i in range(N):
        cursor = N - 1
        for j in range(N - 1, -1, -1):
            if copy_arr[i][j] != 0:
                temp = copy_arr[i][j]
                copy_arr[i][j] = 0

                if copy_arr[i][cursor] == 0:
                    copy_arr[i][cursor] = temp
                
                elif copy_arr[i][cursor] == temp:
                    copy_arr[i][cursor] += temp
                    cursor -=1

                else:
                    cursor -= 1
                    copy_arr[i][cursor] = temp

    return copy_arr

def up(copy_arr):
    for j in range(N):
        cursor = 0
        for i in range(N):
            if copy_arr[i][j] != 0:
                temp = copy_arr[i][j]
                copy_arr[i][j] = 0

                if copy_arr[cursor][j] == 0:
                    copy_arr[cursor][j] = temp
                
                elif copy_arr[cursor][j] == temp:
                    copy_arr[cursor][j] += temp
                    cursor += 1
                
                else:
                    cursor += 1
                    copy_arr[cursor][j] = temp

    return copy_arr

def down(copy_arr):
    for j in range(N):
        cursor = N - 1
        for i in range(N - 1, -1, -1):
            if copy_arr[i][j] != 0:
                temp = copy_arr[i][j]
                copy_arr[i][j] = 0

                if copy_arr[cursor][j] == 0:
                    copy_arr[cursor][j] = temp
                
                elif copy_arr[cursor][j] == temp:
                    copy_arr[cursor][j] += temp
                    cursor -=1

                else:
                    cursor -= 1
                    copy_arr[cursor][j] = temp

    return copy_arr

def dfs(n, arr):
    global res
    if n == 5:
        for i in range(N):
            for j in range(N):
                if arr [i][j] > res:
                    res = arr[i][j]
        return
    for k in range(4):
        copy_arr = copy.deepcopy(arr)
        if k == 0:
            dfs(n + 1, left(copy_arr))
        elif k == 1:
            dfs(n + 1, right(copy_arr))
        elif k == 2:
            dfs(n + 1, up(copy_arr))
        else:
            dfs(n + 1, down(copy_arr))

dfs(0, arr)
print(res)
