import sys
from collections import deque

def bfs(color):
    cnt = 1
    while que:
        q = que.popleft()
        for k in range(4):
            ni = dr[k] + q[0]
            nj = dc[k] + q[1]
            if 0 <= ni < N and 0 <= nj < M and visited[ni][nj] == 0 and battle[ni][nj] == color:
                visited[ni][nj] = 1
                que.append([ni, nj])
                cnt += 1
    return cnt ** 2

N, M = map(int, sys.stdin.readline().rstrip().split())
battle = [list(sys.stdin.readline().rstrip()) for _ in range(N)]

dr, dc = [-1, 0, 1, 0], [0, 1, 0, -1]
visited = [[0] * M for _ in range(N)]

que = deque()

w = 0
b = 0
for i in range(N):
    for j in range(M):
        if visited[i][j] == 0:
            color = battle[i][j]
            que.append([i, j])
            visited[i][j] = 1
            if color == 'W':
                w += bfs(color)
            else:
                b += bfs(color)
print(w, b)
