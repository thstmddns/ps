    print(sum(li[:5]))

    min1 = li_m[0]
    min2 = li_m[0] + li_m[1]
    min3 = li_m[0] + li_m[1] + li_m[2]

    li_c1 = (N - 2) ** 2 + 4 * (N - 1) * (N - 2)
    li_c2 = (N - 1) * 4 + 4 * (N - 2)
