import sys

N = int(sys.stdin.readline())
li = []
for _ in range(N):
    s = sys.stdin.readline()
    li.append(s[0])
set_li = set(li)
ss = ''
for s in set_li:
    if li.count(s) >= 5:
        ss += s
if len(ss) == 0:
    print('PREDAJA')
else:
    ss = sorted(ss)
    for sss in ss:
        print(sss, end='')
