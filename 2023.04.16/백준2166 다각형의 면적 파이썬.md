```
import sys

N = int(sys.stdin.readline())
point = [list(map(int, sys.stdin.readline().split())) for _ in range(N)]
point.append(point[0])
x = 0
y = 0
for i in range(N):
    x += point[i][0] * point[i + 1][1]
    y += point[i + 1][0] * point[i][1]
result = round((x - y) * 0.5, 1)

if result < 0:
    result *= -1
print(result)
```
