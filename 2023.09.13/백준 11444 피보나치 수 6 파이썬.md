import sys

def multi_mul(mat1, mat2):
    l = len(mat1)
    v = [[0] * l for _ in range(l)]
    for r in range(l):
        for c in range(l):
            e = 0
            for k in range(l):
                e += mat1[r][k] * mat2[k][c]
            v[r][c] = e % p
    return v

def mul(mat, N):
    if N == 1:
        for i in range(len(mat)):
            for j in range(len(mat)):
                mat[i][j] %= p
        return mat
    temp = mul(mat, N // 2)
    if N % 2 == 1:
        return multi_mul(multi_mul(temp, temp), mat)
    else:
        return multi_mul(temp, temp)

N = int(sys.stdin.readline())
p = 1000000007
mat = [[1, 1], [1, 0]]
print(mul(mat, N)[0][1])
