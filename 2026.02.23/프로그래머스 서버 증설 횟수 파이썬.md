def solution(players, m, k):
    cnt = 0
    server_cnt = [0] * 24
    
    for i in range(24):
        p = players[i] // m
        
#         서버 증설횟수 세기
        if p > server_cnt[i]:
            temp = p - server_cnt[i]
            cnt += temp
#           
            if i + k < 24:
                for t in range(i, i + k):
                    server_cnt[t] += temp
            else:
                for t in range(i, 24):
                    server_cnt[t] += temp
    answer = cnt
    return answer
