```
import sys

# 입력 값 설정
C, N = map(int, sys.stdin.readline().split())
city_information = [list(map(int, sys.stdin.readline().split())) for _ in range(N)]

# dp 생성
dp = [[987654321] * (C + 101) for _ in range(N + 1)]

dp[0][0] = 0

# 배낭 알고리즘 구축
for i in range(1, N + 1):
    for j in range(0, C + 101):
        if city_information[i - 1][1] > j:
            dp[i][j] = dp[i - 1][j]
        else:
            dp[i][j] = min(dp[i - 1][j], dp[i][max(0, j - city_information[i - 1][1])] + city_information[i - 1][0])

print(min(dp[N][C:]))
```
