import sys

N, K = map(int, sys.stdin.readline().split())
caffeine = [0] + list(map(int, sys.stdin.readline().split()))

dp = [[987654321] * (K + 1) for _ in range(N + 1)]

for k in range(N + 1):
    dp[k][0] = 0
    
for i in range(1, N + 1):
    for j in range(1, K + 1):
        if j < caffeine[i]:
            dp[i][j] = dp[i - 1][j]
        else:
            dp[i][j] = min(dp[i - 1][j], dp[i - 1][j-caffeine[i]] + 1)

if dp[N][K] == 987654321:
    print(-1)
else:
    print(dp[N][K])
