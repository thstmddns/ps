import sys

N, T = map(int, sys.stdin.readline().split())
day = [[0, 0]] + [list(map(int, sys.stdin.readline().split())) for _ in range(N)]
total = 0

for t in range(1, N + 1):
    total += day[t][0]
if total <= T:
    print(0)
else:
    dp = [[0] * (T + 1) for _ in range(N + 1)]
    for i in range(1, N + 1):
        for j in range(T + 1):
            if j < day[i][0]:
                dp[i][j] = dp[i - 1][j]
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - day[i][0]] + day[i][1])
    sum_value = sum(day[d][1] for d in range(1, N + 1))
    print(sum_value - dp[N][T])
